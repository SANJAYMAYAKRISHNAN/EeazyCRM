# Check Python Version
python --version

# Upgrade pip
python -m pip install --upgrade pip

# Install Requirements
pip install -r requirements.txt || {
  echo "Installation failed. Please check the error message for details."
  
  # Optional: Install packages individually if required
  # Replace 'package_name' with the specific package causing issues
  # pip install package_name

  # Set up a Virtual Environment
  python -m venv myenv
  source myenv/bin/activate  # On Windows: myenv\Scripts\activate
  pip install -r requirements.txt
}
